 53:     /**
 54:      * Get the base URL of the application.
 55:      *
 56:      * @return string
 57:      */
 58:     public static function base()
 59:     {
 60:         if (isset(static::$base)) return static::$base;
 61: 
 62:         $base = 'http://localhost';
 63: 
 64:         // If the application URL configuration is set, we will just use that
 65:         // instead of trying to guess the URL from the $_SERVER array's host
 66:         // and script variables as this is more reliable.
 67:         if (($url = Config::get('application.url')) !== '')
 68:         {
 69:             $base = $url;
 70:         }
 71:         elseif (isset($_SERVER['HTTP_HOST']))
 72:         {
 73:             $protocol = (Request::secure()) ? 'https://' : 'http://';
 74: 
 75:             // Basically, by removing the basename, we are removing everything after the
 76:             // and including the front controller from the request URI. Leaving us with
 77:             // the path in which the framework is installed.
 78:             $script = $_SERVER['SCRIPT_NAME'];
 79: 
 80:             $path = str_replace(basename($script), '', $script);
 81: 
 82:             // Now that we have the base URL, all we need to do is attach the protocol
 83:             // and the HTTP_HOST to build the full URL for the application. We also
 84:             // trim off trailing slashes to clean the URL.
 85:             $base = rtrim($protocol.$_SERVER['HTTP_HOST'].$path, '/');
 86:         }
 87: 
 88:         return static::$base = $base;
 89:     }

 91:     /**
 92:      * Generate an application URL.
 93:      *
 94:      * <code>
 95:      *      // Create a URL to a location within the application
 96:      *      $url = URL::to('user/profile');
 97:      *
 98:      *      // Create a HTTPS URL to a location within the application
 99:      *      $url = URL::to('user/profile', true);
100:      * </code>
101:      *
102:      * @param  string  $url
103:      * @param  bool    $https
104:      * @return string
105:      */
106:     public static function to($url = '', $https = false)
107:     {
108:         if (filter_var($url, FILTER_VALIDATE_URL) !== false) return $url;
109: 
110:         $root = static::base().'/'.Config::get('application.index');
111: 
112:         // Since SSL is not often used while developing the application, we allow the
113:         // developer to disable SSL on all framework generated links to make it more
114:         // convenient to work with the site while developing locally.
115:         if ($https and Config::get('application.ssl'))
116:         {
117:             $root = preg_replace('~http://~', 'https://', $root, 1);
118:         }
119: 
120:         return rtrim($root, '/').'/'.ltrim($url, '/');
121:     }

134:     /**
135:      * Generate a URL to a controller action.
136:      *
137:      * <code>
138:      *      // Generate a URL to the "index" method of the "user" controller
139:      *      $url = URL::to_action('user@index');
140:      *
141:      *      // Generate a URL to http://example.com/user/profile/taylor
142:      *      $url = URL::to_action('user@profile', array('taylor'));
143:      * </code>
144:      *
145:      * @param  string  $action
146:      * @param  array   $parameters
147:      * @return string
148:      */
149:     public static function to_action($action, $parameters = array())
150:     {
151:         // This allows us to use true reverse routing to controllers, since
152:         // URIs may be setup to handle the action that do not follow the
153:         // typical Laravel controller URI conventions.
154:         $route = Router::uses($action);
155: 
156:         if ( ! is_null($route))
157:         {
158:             return static::explicit($route, $action, $parameters);
159:         }
160:         // If no route was found that handled the given action, we'll just
161:         // generate the URL using the typical controller routing setup
162:         // for URIs and turn SSL to false.
163:         else
164:         {
165:             return static::convention($action, $parameters);
166:         }
167:     }

184:     /**
185:      * Generate an action URI by convention.
186:      *
187:      * @param  string  $action
188:      * @param  array   $parameters
189:      * @return string
190:      */
191:     protected static function convention($action, $parameters)
192:     {
193:         list($bundle, $action) = Bundle::parse($action);
194: 
195:         $bundle = Bundle::get($bundle);
196: 
197:         // If a bundle exists for the action, we will attempt to use it's "handles"
198:         // clause as the root of the generated URL, as the bundle can only handle
199:         // URIs that begin with that string.
200:         $root = $bundle['handles'] ?: '';
201: 
202:         $https = false;
203: 
204:         $parameters = implode('/', $parameters);
205: 
206:         // We'll replace both dots and @ signs in the URI since both are used
207:         // to specify the controller and action, and by convention should be
208:         // translated into URI slashes.
209:         $uri = $root.'/'.str_replace(array('.', '@'), '/', $action);
210: 
211:         $uri = static::to(str_finish($uri, '/').$parameters);
212: 
213:         return trim($uri, '/');
214:     }

226:

228:

240:     /**
241:      * Generate a URL from a route name.
242:      *
243:      * <code>
244:      *      // Create a URL to the "profile" named route
245:      *      $url = URL::to_route('profile');
246:      *
247:      *      // Create a URL to the "profile" named route with wildcard parameters
248:      *      $url = URL::to_route('profile', array($username));
249:      * </code>
250:      *
251:      * @param  string  $name
252:      * @param  array   $parameters
253:      * @param  bool    $https
254:      * @return string
255:      */
256:     public static function to_route($name, $parameters = array())
257:     {
258:         if (is_null($route = Routing\Router::find($name)))
259:         {
260:             throw new \Exception("Error creating URL for undefined route [$name].");
261:         }
262: 
263:         // To determine whether the URL should be HTTPS or not, we look for the "https"
264:         // value on the route action array. The route has control over whether the URL
265:         // should be generated with an HTTPS protocol string or just HTTP.
266:         $https = array_get(current($route), 'https', false);
267: 
268:         $uri = trim(static::transpose(key($route), $parameters), '/');
269: 
270:         return static::to($uri, $https);
271:     }
272: 
273:     /**
274:      * Substitute the parameters in a given URI.
275:      *
276:      * @param  string  $uri
277:      * @param  array   $parameters
278:      * @return string
279:      */
280:     public static function transpose($uri, $parameters)
281:     {
282:         // Spin through each route parameter and replace the route wildcard segment
283:         // with the corresponding parameter passed to the method. Afterwards, we'll
284:         // replace all of the remaining optional URI segments.
285:         foreach ((array) $parameters as $parameter)
286:         {
287:             if ( ! is_null($parameter))
288:             {
289:                 $uri = preg_replace('/\(.+?\)/', $parameter, $uri, 1);
290:             }
291:         }
292: 
293:         // If there are any remaining optional place-holders, we'll just replace
294:         // them with empty strings since not every optional parameter has to be
295:         // in the array of parameters that were passed.
296:         $uri = str_replace(array_keys(Router::$optional), '', $uri);
297: 
298:         return trim($uri, '/');
299:     }
