 38:     /**
 39:      * Run a database migration command.
 40:      *
 41:      * @param  array  $arguments
 42:      * @return void
 43:      */
 44:     public function run($arguments = array())
 45:     {
 46:         // If no arguments were passed to the task, we will just migrate
 47:         // to the latest version across all bundles. Otherwise, we will
 48:         // parse the arguments to determine the bundle for which the
 49:         // database migrations should be run.
 50:         if (count($arguments) == 0)
 51:         {
 52:             $this->migrate();
 53:         }
 54:         else
 55:         {
 56:             $this->migrate(array_get($arguments, 0));
 57:         }
 58:     }

 60:     /**
 61:      * Run the outstanding migrations for a given bundle.
 62:      *
 63:      * @param  string  $bundle
 64:      * @param  int     $version
 65:      * @return void
 66:      */
 67:     public function migrate($bundle = null, $version = null)
 68:     {
 69:         $migrations = $this->resolver->outstanding($bundle);
 70: 
 71:         if (count($migrations) == 0)
 72:         {
 73:             echo "No outstanding migrations.";
 74: 
 75:             return;
 76:         }
 77: 
 78:         // We need to grab the latest batch ID and increment it by one.
 79:         // This allows us to group the migrations so we can easily
 80:         // determine which migrations need to roll back.
 81:         $batch = $this->database->batch() + 1;
 82: 
 83:         foreach ($migrations as $migration)
 84:         {
 85:             $migration['migration']->up();
 86: 
 87:             echo 'Migrated: '.$this->display($migration).PHP_EOL;
 88: 
 89:             // After running a migration, we log its execution in the migration
 90:             // table so that we can easily determine which migrations we'll
 91:             // reverse in the event of a migration rollback.
 92:             $this->database->log($migration['bundle'], $migration['name'], $batch);
 93:         }
 94:     }
 95: 
 96:     /**
 97:      * Rollback the latest migration command.
 98:      *
 99:      * @param  array  $arguments
100:      * @return bool
101:      */
102:     public function rollback($arguments = array())
103:     {
104:         $migrations = $this->resolver->last();
105: 
106:         if (count($migrations) == 0)
107:         {
108:             echo "Nothing to rollback.";
109: 
110:             return false;
111:         }
112: 
113:         // The "last" method on the resolver returns an array of migrations,
114:         // along with their bundles and names. We will iterate through each
115:         // migration and run the "down" method.
116:         foreach (array_reverse($migrations) as $migration)
117:         {
118:             $migration['migration']->down();
119: 
120:             echo 'Rolled back: '.$this->display($migration).PHP_EOL;
121: 
122:             // By only removing the migration after it has successfully rolled back,
123:             // we can re-run the rollback command in the event of any errors with
124:             // the migration and pick up where we left off.
125:             $this->database->delete($migration['bundle'], $migration['name']);
126:         }
127: 
128:         return true;
129:     }

143:      * Install the database tables used by the migration system.
144:      *
145:      * @return void
146:      */
147:     public function install()
148:     {
149:         Schema::table('laravel_migrations', function($table)
150:         {
151:             $table->create();
152: 
153:             // Migrations can be run for a specific bundle, so we'll use
154:             // the bundle name and string migration name as an unique ID
155:             // for the migrations, allowing us to easily identify which
156:             // migrations have been run for each bundle.
157:             $table->string('bundle');
158: 
159:             $table->string('name');
160: 
161:             // When running a migration command, we will store a batch
162:             // ID with each of the rows on the table. This will allow
163:             // us to grab all of the migrations that were run for the
164:             // last command when performing rollbacks.
165:             $table->integer('batch');
166: 
167:             $table->primary(array('bundle', 'name'));
168:         });
169: 
170:         echo "Migration table created successfully.";
171:     }

173:     /**
174:      * Generate a new migration file.
175:      *
176:      * @param  array   $arguments
177:      * @return string
178:      */
179:     public function make($arguments = array())
180:     {
181:         if (count($arguments) == 0)
182:         {
183:             throw new \Exception("I need to know what to name the migration.");
184:         }
185: 
186:         list($bundle, $migration) = Bundle::parse($arguments[0]);
187: 
188:         // The migration path is prefixed with the date timestamp, which
189:         // is a better way of ordering migrations than a simple integer
190:         // incrementation, since developers may start working on the
191:         // next migration at the same time unknowingly.
192:         $prefix = date('Y_m_d_His');
193: 
194:         $path = Bundle::path($bundle).'migrations'.DS;
195: 
196:         // If the migration directory does not exist for the bundle,
197:         // we will create the directory so there aren't errors when
198:         // when we try to write the migration file.
199:         if ( ! is_dir($path)) mkdir($path);
200: 
201:         $file = $path.$prefix.'_'.$migration.EXT;
202: 
203:         File::put($file, $this->stub($bundle, $migration));
204: 
205:         echo "Great! New migration created!";
206: 
207:         // Once the migration has been created, we'll return the
208:         // migration file name so it can be used by the task
209:         // consumer if necessary for futher work.
210:         return $file;
211:     }

213:     /**
214:      * Get the stub migration with the proper class name.
215:      *
216:      * @param  string  $bundle
217:      * @param  string  $migration
218:      * @return string
219:      */
220:     protected function stub($bundle, $migration)
221:     {
222:         $stub = File::get(path('sys').'cli/tasks/migrate/stub'.EXT);
223: 
224:         $prefix = Bundle::class_prefix($bundle);
225: 
226:         // The class name is formatted simialrly to tasks and controllers,
227:         // where the bundle name is prefixed to the class if it is not in
228:         // the default "application" bundle.
229:         $class = $prefix.Str::classify($migration);
230: 
231:         return str_replace('{{class}}', $class, $stub);
232:     }
