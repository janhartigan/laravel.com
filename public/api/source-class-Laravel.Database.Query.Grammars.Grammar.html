 29:     /**
 30:      * Generate the SQL for every component of the query.
 31:      *
 32:      * @param  Query  $query
 33:      * @return array
 34:      */
 35:     final protected function components($query)
 36:     {
 37:         // Each portion of the statement is compiled by a function corresponding
 38:         // to an item in the components array. This lets us to keep the creation
 39:         // of the query very granular and very flexible.
 40:         //
 41:         // Note that each component also connects to a public property on the
 42:         // query instance, allowing us to pass the correct data into each
 43:         // of the compiler functions.
 44:         foreach ($this->components as $component)
 45:         {
 46:             if ( ! is_null($query->$component))
 47:             {
 48:                 $sql[$component] = call_user_func(array($this, $component), $query);
 49:             }
 50:         }
 51: 
 52:         return (array) $sql;
 53:     }

 55:     /**
 56:      * Concatenate an array of SQL segments, removing those that are empty.
 57:      *
 58:      * @param  array   $components
 59:      * @return string
 60:      */
 61:     final protected function concatenate($components)
 62:     {
 63:         return implode(' ', array_filter($components, function($value)
 64:         {
 65:             return (string) $value !== '';
 66:         }));
 67:     }

110:     /**
111:      * Compile the JOIN clauses for a query.
112:      *
113:      * @param  Query   $query
114:      * @return string
115:      */
116:     protected function joins(Query $query)
117:     {
118:         // We need to iterate through each JOIN clause that is attached to the
119:         // query an translate it into SQL. The table and the columns will be
120:         // wrapped in identifiers to avoid naming collisions.
121:         //
122:         // Once all of the JOINs have been compiled, we can concatenate them
123:         // together using a single space, which should give us the complete
124:         // set of joins in valid SQL that can appended to the query.
125:         foreach ($query->joins as $join)
126:         {
127:             $table = $this->wrap_table($join->table);
128: 
129:             $clauses = array();
130: 
131:             // Each JOIN statement may have multiple clauses, so we will
132:             // iterate through each clause creating the conditions then
133:             // we will concatenate them all together.
134:             foreach ($join->clauses as $clause)
135:             {
136:                 extract($clause);
137: 
138:                 $column1 = $this->wrap($column1);
139: 
140:                 $column2 = $this->wrap($column2);
141: 
142:                 $clauses[] = "{$connector} {$column1} {$operator} {$column2}";
143:             }
144: 
145:             // The first clause will have a connector on the front,
146:             // but it is not needed on the first condition, so we
147:             // will strip it off of the condition before adding
148:             // it to the array of joins.
149:             $search = array('AND ', 'OR ');
150: 
151:             $clauses[0] = str_replace($search, '', $clauses[0]);
152: 
153:             $clauses = implode(' ', $clauses);
154: 
155:             $sql[] = "{$join->type} JOIN {$table} ON {$clauses}";
156:         }
157: 
158:         // Finally, we should have an array of JOIN clauses
159:         // that we can implode together and return as the
160:         // complete SQL for the JOIN of the query.
161:         return implode(' ', $sql);
162:     }

164:     /**
165:      * Compile the WHERE clause for a query.
166:      *
167:      * @param  Query   $query
168:      * @return string
169:      */
170:     final protected function wheres(Query $query)
171:     {
172:         if (is_null($query->wheres)) return '';
173: 
174:         // Each WHERE clause array has a "type" that is assigned by the query
175:         // builder, and each type has its own compiler function. We will call
176:         // the appropriate compiler for each where clause in the query.
177:         //
178:         // Keeping each particular where clause in its own "compiler" allows
179:         // us to keep the query generation process very granular, making it
180:         // easier to customize derived grammars for other databases.
181:         foreach ($query->wheres as $where)
182:         {
183:             $sql[] = $where['connector'].' '.$this->{$where['type']}($where);
184:         }
185: 
186:         if  (isset($sql))
187:         {
188:             // We attach the boolean connector to every where segment just
189:             // for convenience. Once we have built the entire clause we'll
190:             // remove the first instance of a connector from the clause.
191:             return 'WHERE '.preg_replace('/AND |OR /', '', implode(' ', $sql), 1);
192:         }
193:     }

329:     /**
330:      * Compile a SQL INSERT statment from a Query instance.
331:      *
332:      * This method handles the compilation of single row inserts and batch inserts.
333:      *
334:      * @param  Query   $query
335:      * @param  array   $values
336:      * @return string
337:      */
338:     public function insert(Query $query, $values)
339:     {
340:         $table = $this->wrap_table($query->from);
341: 
342:         // Force every insert to be treated like a batch insert. This simply makes
343:         // creating the SQL syntax a little easier on us since we can always treat
344:         // the values as if it is an array containing multiple inserts.
345:         if ( ! is_array(reset($values))) $values = array($values);
346: 
347:         // Since we only care about the column names, we can pass any of the insert
348:         // arrays into the "columnize" method. The columns should be the same for
349:         // every insert to the table so we can just use the first record.
350:         $columns = $this->columnize(array_keys(reset($values)));
351: 
352:         // Build the list of parameter place-holders of values bound to the query.
353:         // Each insert should have the same number of bound paramters, so we can
354:         // just use the first array of values.
355:         $parameters = $this->parameterize(reset($values));
356: 
357:         $parameters = implode(', ', array_fill(0, count($values), "($parameters)"));
358: 
359:         return "INSERT INTO {$table} ({$columns}) VALUES {$parameters}";
360:     }

362:     /**
363:      * Compile a SQL UPDATE statment from a Query instance.
364:      *
365:      * @param  Query   $query
366:      * @param  array   $values
367:      * @return string
368:      */
369:     public function update(Query $query, $values)
370:     {
371:         $table = $this->wrap_table($query->from);
372: 
373:         // Each column in the UPDATE statement needs to be wrapped in keyword
374:         // identifiers, and a place-holder needs to be created for each value
375:         // in the array of bindings. Of course, if the value of the binding
376:         // is an expression, the expression string will be injected.
377:         foreach ($values as $column => $value)
378:         {
379:             $columns[] = $this->wrap($column).' = '.$this->parameter($value);
380:         }
381: 
382:         $columns = implode(', ', $columns);
383: 
384:         // UPDATE statements may be constrained by a WHERE clause, so we'll
385:         // run the entire where compilation process for those contraints.
386:         // This is easily achieved by passing the query to the "wheres"
387:         // method which will call all of the where compilers.
388:         return trim("UPDATE {$table} SET {$columns} ".$this->wheres($query));
389:     }

407:     /**
408:      * Transform an SQL short-cuts into real SQL for PDO.
409:      *
410:      * @param  string  $sql
411:      * @param  array   $bindings
412:      * @return string
413:      */
414:     public function shortcut($sql, $bindings)
415:     {
416:         // Laravel provides an easy short-cut notation for writing raw
417:         // WHERE IN statements. If (...) is in the query, it will be
418:         // replaced with the correct number of parameters based on
419:         // the bindings for the query.
420:         if (strpos($sql, '(...)') !== false)
421:         {
422:             for ($i = 0; $i < count($bindings); $i++)
423:             {
424:                 // If the binding is an array, we can just assume it's
425:                 // used to fill a "where in" condition, so we'll just
426:                 // replace the next place-holder in the query.
427:                 if (is_array($bindings[$i]))
428:                 {
429:                     $parameters = $this->parameterize($bindings[$i]);
430: 
431:                     $sql = preg_replace('~\(\.\.\.\)~', "({$parameters})", $sql, 1);
432:                 }
433:             }           
434:         }
435: 
436:         return trim($sql);
437:     }
